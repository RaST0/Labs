	int O[1][3] = { { 3,2,1 } };

	int P[1][3] = { { 4,3,2 } };

	int S[1][3] = { { 0,0,0 } };

	std::cout << std::endl;


	for (int i = 0; i < 1; i++) 
	{                                 // проходим по всем элементам финального вектора
		for (int j = 0; j < 3; j++) 
		{
			S[i][j] = 0;

			{
				S[i][j] += O[i][j] + P[i][j]; // производим сложение двух векторов
			}
			std::cout << S[i][j] << " " << std::endl;
		}
	}
	std::cout << std::endl;
